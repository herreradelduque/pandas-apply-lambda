    "    n = int(sum([ord(i) for i in x]) / y)\n",
    "    \n",
    "    if n == 0:\n",
    "        return False\n",
    "    \n",
    "    for i in range(2,n):\n",
    "        if (n%i) == 0:\n",
    "            return False\n",
    "        \n",
    "    return True"
